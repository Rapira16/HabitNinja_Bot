  83def init_db():

  88    """
  89    conn = sqlite3.connect("habits.db")
  90    c = conn.cursor()
  91
  92    c.execute(
  93        """CREATE TABLE IF NOT EXISTS users
  94                 (user_id INTEGER PRIMARY KEY,
  95                  name TEXT,
  96                  motivation_time TEXT,
  97                  last_motivation INTEGER)"""
  98    )
  99
 100    c.execute(
 101        """CREATE TABLE IF NOT EXISTS habits
 102                 (id INTEGER PRIMARY KEY AUTOINCREMENT,
 103                  user_id INTEGER,
 104                  habit_name TEXT,
 105                  created_date TEXT,
 106                  count INTEGER DEFAULT 0,
 107                  UNIQUE(user_id, habit_name))"""
 108    )
 109
 110    c.execute(
 111        """CREATE TABLE IF NOT EXISTS reminders
 112                 (id INTEGER PRIMARY KEY AUTOINCREMENT,
 113                  user_id INTEGER,
 114                  habit_id INTEGER UNIQUE,
 115                  reminder_time TEXT,
 116                  last_reminded INTEGER)"""
 117    )
